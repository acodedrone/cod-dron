
class DBFactory(ABC):
    """
    Абстрактный класс, представляющий фабрику для подключения к базе данных.

    Методы:
    connect(path_to_db: str):
        Абстрактный метод для подключения к базе данных.
    """


class SQLiteDBFactory(DBFactory):
    """
    Реализация фабрики для подключения к SQLite.

    Методы:
    connect(path_to_db: str):
        Подключает к базе данных SQLite и возвращает соединение.
    """
    def connect(self, path_to_db: str):
        """
        Подключает к базе данных SQLite.

        Аргументы:
        path_to_db (str): Путь к базе данных SQLite.

        Возвращаемое значение:
        conn (sqlite3.Connection): Соединение с базой данных SQLite.
        None: Если произошла ошибка подключения.

        Описание:
        - Пытается подключиться к базе данных SQLite.
        - Логирует успешное подключение.
        - Возвращает соединение или None в случае ошибки.

        Возможные исключения:
        - sqlite3.Error: Обрабатывает ошибки подключения к базе данных.
        """

class PostgreSQLDBFactory(DBFactory):

    Методы:
    connect(path_to_db: str):
        Подключает к базе данных PostgreSQL и возвращает True.
    """
    def connect(self, path_to_db: str):
        """
        Подключает к базе данных PostgreSQL.

        Аргументы:
        path_to_db (str): Путь к базе данных PostgreSQL.

        Возвращаемое значение:
        True: Всегда возвращает True для демонстрации.

        Описание:
        - Логирует успешное подключение.
        """
